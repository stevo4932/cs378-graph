121  friend std::pair<edge_descriptor, bool> edge (vertex_descriptor u, vertex_descriptor v, const Graph& graph) {
122  // trys to find the matching edge.
123  for(edges_size_type i = 0; i < graph.e_size; ++i){
124  std::vector<vertex_descriptor> vec = graph.e_list.at((int)i);
125  assert(vec.size() == 2);
126  if(vec.at(0) == u && vec.at(1) == v){
127  return std::make_pair(edge_descriptor(i), true);
128  }
